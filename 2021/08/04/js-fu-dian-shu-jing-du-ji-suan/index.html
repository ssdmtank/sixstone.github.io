js浮点数精度计算

精度丢失
javascript基础
发布日期:   2021-08-04
更新日期:   2021-08-05
文章字数:   1.1k
阅读时长:   5 分
阅读次数:  

问题

浮点数计算精度不准确

// 加法
0.1 + 0.2 = 0.30000000000000004
// 减法
0.3 - 0.2 = 0.09999999999999998
// 乘法
0.3 * 3 = 0.8999999999999999
// 除法
0.3 / 0.1 = 2.9999999999999996
// 保留小数
3.335.toFixed(20) = "3.33499999999999996447"

原因

  • 在计算机角度，计算机算的是二进制，而不是十进制。
  • 二进制后变成了无线不循环的数，而计算机可支持浮点数的小数部分可支持到52位。
十进制 二进制
0.1 0.0001 1001 1001 1001 …
0.2 0.0011 0011 0011 0011 …
0.3 0.0100 1100 1100 1100 …
0.4 0.0110 0110 0110 0110 …
0.5 0.1
0.6 0.1001 1001 1001 1001 …

小数转二进制的方法： 对小数点以后的数乘以2，有一个结果吧，取结果的整数部分（不是1就是0），然后再用小数部分再乘以2，再取结果的整数部分……以此类推，直到小数部分为0或者位数已经够了。

演示： 

0.125 × 2 = 0.25 .......................0
0.25  × 2 = 0.5  .......................0
0.5   × 2 = 1.0  .......................1

验证： 0.001 => 0×2的0次方+0×2的-1次方+0×2的-2次方+0×2的-3次方=1/8=0.125，正确。

0.1不能转化二进制

0.1 × 2 = 0.2 ......................0
0.2 × 2 = 0.4 ......................0
0.4 × 2 = 0.8 ......................0
0.8 × 2 = 1.6.......................1
0.6 × 2 = 1.2.......................1
0.2 × 2 = 0.4.......................0
.....

因为不能得到小数位为0且位数会溢出
所有0.1转化会无限循环 => 0.0001 1001 1001 1001 …

toFixed四舍五入不准确性

1.595.toFixed(2)   => "1.59"
1.5952.toFixed(2)  => "1.60"

按照上图步骤演示1.595.toFixed(2) => “1.59”下处理过程:

  • x为1.595，小于$10^{21}$，f为2，要使n➗ $10^f$ - x准确的数学值尽可能接近零，取n为159和n为160，
  • 159/100-1.595 = -0.004999999999999893
  • 160/100-1.595 = 0.0050000000000001155
  • 可以看到最接近0的是 -0.004999999999999893 故n为159，m为159，k为3
    k−f为1，故a为15，则b为9，所以(1.595).toFixed(1)的结果为1.59
  • 159/100-1.595 不为1.595, 原理和0.1不能转化二进制一样

解决方案

方法1 Math.js

  1. 安装
    npm install mathjs / yarn add mathjs
  2. 封装
    import * as math from 'mathjs';
export default {
    // 加
	add(num1,num2){
		return math.add(math.bignumber(num1),math.bignumber(num2));
	},
	// 乘
	multiply(num1,num2){
		return math.multiply(math.bignumber(num1),math.bignumber(num2));
	},
	// 减
	subtract(num1,num2){
		return math.subtract(math.bignumber(num1),math.bignumber(num2));
	},
	// 除
	divide(num1,num2){
		return math.divide(math.bignumber(num1),math.bignumber(num2));
	}
}

方法2 浮点数转化整数计算

/** 判断是否为整数 */
const isInteger = (obj) => Math.floor(obj) === obj

/**
* 将浮点数转换为整数
* @ param floatNum 浮点数
* @ return { time //倍数, num // 整数}
*/
const floatToInteger = (num) => {
    let times = 1
    // 整数不处理
    if(isInteger(num)) return { times, num}
    // 倍数
    times = 10 ** `${num}`.split('.')[1].length
    // 转换整数
    const intNum = parseInt(Math.abs(num) * times + 0.5, 10)
    return {times, num: num < 0 ? -intNum : intNum}
}

/**
 * 实现加减乘除运算，确保不丢失精度
 * @param num1 {number} 运算数1
 * @param num2 {number} 运算数2
 * @param digits {number} 精度，保留的小数点数，比如 2, 即保留为两位小数
 * @param type {string} 运算类型，有加减乘除（add/subtract/multiply/divide）
 */
const math = (num1, num2, digits, type) => {
    // 先拆解
    const { times : t1, num : n1} = floatToInteger(num1)
    const { times : t2, num : n2} = floatToInteger(num2)
    const max = t1 > t2 ? t1 : t2
    let result = null

    switch (type) {
        case 'add':
            if(t1 === t2){ // 两个小数位数相同
                result = n1 + n2
            }else if(t1 > t2){ // num1 小数位 大于 num2
                result = n1 + n2 * (t1 / t2)
            }else{ // num1 小数位 小于 num2
                result = n1 * (t2 / t1) + n2
            }
            return result / max
        case 'subtract':
            if (t1 === t2) {
                result = n1 - n2
            } else if (t1 > t2) {
                result = n1 - n2 * (t1 / t2)
            } else {
                result = n1 * (t2 / t1) - n2
            }
            return result / max
        case 'multiply':
             return (n1 * n2) / (t1 * t2)
        case 'divide':
            return (n1 / n2) * (t2 / t1)
    }
}

const add = (num1 , num2 , digits) => math(num1, num2, digits, 'add')
const subtract = (num1 , num2 , digits) => math(num1, num2, digits, 'subtract')
const multiply = (num1 , num2 , digits) => math(num1, num2, digits, 'multiply')
const divide = (num1 , num2 , digits) => math(num1, num2, digits, 'divide')
const toFixed = (num, s) => {
    if (typeof num !== 'number') throw new Error("number不是数字")
    // 修复toFixed精度问题
    let result = Math.round(10 ** s * num) / 10 ** s
    // 补零
    result = String(result)
    if (result.indexOf('.') === -1){ // 整数
        if(s !== 0){
            result += '.'
            result += new Array(s + 1).join('0')
        }
    } else {
        let arr = result.split('.');
        if (arr[1].length < s) {
            arr[1] += new Array(s - arr[1].length + 1).join('0')
        }
        result = arr.join('.')
    }
    return result
}

export const $math =  {
  add,
  subtract, 
  multiply,
  divide,
  toFixed
}
